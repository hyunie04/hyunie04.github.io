### 1. 정수의 출력
---
정수를 출력할 때는 %d 서식지정자를 사용합니다. 
~~~c
int main()
{
  printf("34+56\n");
  printf("%d\n", 34+56);
  printf("%d + %d = %d\n", 34, 56, 34+56);
  printf("34-56\n");
  printf("%d\n", 34-56);
  printf("%d - %d = %d\n", 34, 56, 34-56);
  printf("34*56\n");
  printf("%d\n", 34*56);
  printf("%d * %d = %d\n", 34, 56, 34*56);
  printf("34/56\n");
  printf("%d\n", 34/56);
  printf("%d / %d = %d\n", 34, 56, 34/56);
  return 0;
}
~~~
[출력결과]
34+56  
90  
34 + 56 = 90  
34-56  
-22  
34 - 56 = -22  
34*56  
1904  
34 * 56 = 1904  
34/56  
0  
34 / 56 = 0
